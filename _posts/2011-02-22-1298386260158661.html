 1 <?php
 2 /*
 3    18、 请使用一句话将类似“1910-09-09”的日期各式转换为“一九一零年九月九日”这样的各式。(需要有通用性，是不是一句话不是关键，但不允许使用if\switch\while\for等语句)。
 4  */
 5 $pattern_year = array("/0/", "/1/", "/2/", "/3/", "/4/", "/5/",
 6                 "/6/", "/7/", "/8/", "/9/");
 7 $replace_year = array("零", "一", "二", "三", "四", "五",
 8                 "六", "七", "八", "九");
 9
10 $pattern_month = array("/01/", "/02/", "/03/", "/04/", "/05/",
11                 "/06/", "/07/", "/08/", "/09/", "/10/", "/11/", "/12/");
12 $replace_month = array("一", "二", "三", "四", "五",
13                 "六", "七", "八", "九", "十", "十一", "十二");
14
15 $pattern_day = array("/01/", "/02/", "/03/", "/04/", "/05/",
16                 "/06/", "/07/", "/08/", "/09/", "/10/", "/11/", "/12/",
17                 "/13/", "/14/", "/15/", "/16/", "/17/", "/18/", "/19/",
18                 "/20/", "/21/", "/22/", "/23/", "/24/", "/25/", "/26/",
19                 "/27/", "/28/", "/29/", "/30/", "/31/");
20 $replace_day = array("一", "二", "三", "四", "五",
21                 "六", "七", "八", "九", "十", "十一", "十二",
22                 "十三", "十四", "十五", "十六", "十七", "十八",
23                 "十九", "二十", "二十一", "二十二", "二十三", "二十四", "二十五",
24                 "二十六", "二十七", "二十八", "二十九", "三十", "三十一");
25
26 $str = "1910-09-09";
27 $arr = explode('-', $str);
28
29 $year = preg_replace($pattern_year, $replace_year, $arr[0]);
30 $month = preg_replace($pattern_month, $replace_month, $arr[1]);
31 $day = preg_replace($pattern_day, $replace_day, $arr[2]);
32
33 print $year. "年" . $month . "月" . $day . "日\n";
34 ?>
