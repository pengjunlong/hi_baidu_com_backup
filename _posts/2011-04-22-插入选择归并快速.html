#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <string.h>

#define random(x) (rand()%x)
#define LEN 6666

void insertion_sort(int a[])
{
        int i, j;
        int key;
        for (i = 1; i < LEN; i++)
        {
                key = a[i];
                for (j = i-1; j >= 0&&a[j]>key; j--)
                        a[j+1] = a[j];
                a[j+1] = key;
        }
        for (i = 0; i < LEN; i++)
                printf("%d\n", a[i]);

}

void select_sort(int a[])
{
        int i, j;
        int key, temp;
        for(i = 0; i<LEN; i++)
        {
                key = i;
                for (j = i; j<LEN; j++)
                        if(a[j]<a[key])
                                key = j;
                temp = a[key];
                a[key] = a[i];
                a[i] = temp;
        }
        for (i = 0; i < LEN; i++)
                printf("%d\n", a[i]);
}

void merge(int a[], int start, int mid, int end)
{
        int left_len = mid - start + 1;
        int right_len = end - mid;
        int left[left_len];
        int right[right_len];
        int i, j ,k;

        for (i=0; i<left_len; i++)
                left[i] = a[start+i];
        for (i=0; i<right_len; i++)
                right[i] = a[mid+1+i];
        i=0;
        j=0;
        k=start;
        while(i<left_len && j<right_len)
        {
                if(left[i] < right[j])
                {
                        a[k] = left[i];
                        i++;
                }
                else
                {
                        a[k] = right[j];
                        j++;
                }
                k++;
        }
        if (i == left_len)
        {
                for(; j<right_len; j++, k++)
                        a[k] = right[j];
        }
        else
        {
                for(; i<left_len; i++, k++)
                        a[k] = left[i];
        }
}
void merge_sort(int a[], int start, int end)
{
        if (start < end)
        {
                int mid = (start + end)/2;
                merge_sort(a, start, mid);
                merge_sort(a, mid+1, end);
                merge(a, start, mid, end);
        }
}

int partition(int a[], int start, int end)
{
        int i=start, j=end;
        int pivot=a[start];
        int temp;
        while(i<j)
        {
                for (; a[j]>=pivot&&i<j; j--);
                if(i<j)
                {
                        temp = a[j];
                        a[j] = a[i];
                        a[i] = temp;
                        i++;
                }
                for (; a[i]<pivot&&i<j; i++);
                if(i<j)
                {
                        temp = a[i];
                        a[i] = a[j];
                        a[j] = temp;
                        j--;
                }
        }
        return i;

}

void quick_sort(int a[], int start, int end)
{
        if(start<end)
        {
                int mid = partition(a, start, end);
                quick_sort(a, start, mid-1);
                quick_sort(a, mid+1, end);
        }
}

int main(int argc, char *argv[])
{
        int a[LEN];
        int i;
        srand((int)time(0));
        for (i = 0; i < LEN; i++)
                a[i] = random(100);
        if(argc == 2)
        {
                if (strcmp(argv[1], "insert") == 0)
                        insertion_sort(a);
                if (strcmp(argv[1], "select") == 0)
                        select_sort(a);
                if (strcmp(argv[1], "merge") == 0)
                        merge_sort(a, 0 ,LEN-1);
                if (strcmp(argv[1], "quick") == 0)
                        quick_sort(a, 0, LEN-1);

                for (i = 0; i < LEN; i++)
                        printf("%d\n", a[i]);
        }
        return 0;
}
