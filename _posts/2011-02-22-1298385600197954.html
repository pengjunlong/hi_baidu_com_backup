 6 function mutilSort($data,$key,$sort = 'desc',$type)

 8         $tmp=array();
 9         while(list($k,$v)=each($data)){

11         }

13                 if ($type == "num"){
14                         array_multisort($tmp,SORT_DESC, SORT_NUMERIC,$data);

18         }else{

22                         array_multisort($tmp,SORT_ASC, SORT_STRING,$data);
23                 }
24         }
