 14   无
 15   功能：
 16   将当前已经完成运算的运算符消去，同时将数值数组的位置调整以进行下一次运算。

 21  ***************************************************************************/
 22
 23 void  move(char *f, double *s,int p)  
 24 {
 25         int i=0,len=strlen(f);
 26         for(i=p; i<len; i++)   /*将已经运算过的符号，空出来的位置用后面的符号来填充，*/      
 27         {      /*即把乘和除号的位置用后面的加和减号填充*/
 28                 f[i]=f[i+1];
 29                 s[i]=s[i+1];
 30         }
 31         s[i]=s[i+1];

 34 /**************************************************************************
 35   double convnum(char *c)
 36   输入参数：
 37   char *c :由数字和小数点组成的字符，用以转换成double型的数值。
 38   返回参数：
 39 num:返回转换好的值。
 40 功能：
 41 将输入的字符串先将其小数点以前的部分复制到temp[]数组中，
 42 若有小数点，则将小数点之后的数值，也就是小数部分先进行计算,值存入num中
 43 计算完成后，再对整数部分进行计算，值加上小数部分的值，存入num中。
 44  ***************************************************************************/
 45 double convnum(char *c)
 46 {
 47         double num=0.0;
 48         double a=1.0;
 49         int i=0,p=0,len=0;
 50         char temp[100];
 51         int tempi=0;
 52         int start=0;
 53         int f=1;   /*正负符号指示器，若为1则为正数，为－1，此数为负数*/
 54
 55         len=strlen(c);
 56
 57         if(c[0]=='-')
 58         {
 59                 start=1;
 60                 f=-1;
 61         }
 62         for(i=start; i<len; i++)
 63         {
 64                 if(c[i]=='.')
 65                 {
 66                         p=i;
 67                         break;
 68                 }
 69                 temp[tempi++]=c[i];  /*将整数部分复制到temp[]中*/
 70         }
 71         temp[tempi]='\0';
 72
 73         if(p!=0)
 74         {
 75                 for(i=p+1;i<len;i++) /*将小数部分计算出来*/
 76                 {
 77                         if(c[i]=='.')  /*如果有多余的小数点，则表示输入错误*/
 78                         {
 79                                 //printf("there is more that one dot '.' in number!error!\n");
 80                                 exit(0);
 81                         }
 82                         a=a*0.1;
 83                         num+=(a*(c[i]-48));
 84                 }
 85         }
 86
 87         a=1.0;
 88
 89         len=strlen(temp);           /*计算整数部分*/
 90         for(i=len-1;i>=0; i--)
 91         {
 92                 num=num+(a*(temp[i]-48));
 93                 a*=10;
 94         }
 95
 96         num=num*f;
 97         return num;
 98 }
