 5 //采用倒推算法。
 6 function getTotal($day){
 7         $total=0;
 8         if ($day==1){ //第一天只有一个
 9                 $total = 1;
10         }else{ //以后每一天为前一天的两加上1个。
11                 $total = 2*(getTotal($day-1)+1);
12         }
13         return $total;
14 }
