#
# Use name-based virtual hosting.
#
NameVirtualHost *:80

#
# VirtualHost example:
# Almost any Apache directive may go into a VirtualHost container.
# The first VirtualHost section is used for all requests that do not
# match a ServerName or ServerAlias in any <VirtualHost> block.
#
<VirtualHost *:80>
        ServerName *
        DocumentRoot /usr/local/apache2/htdocs
</VirtualHost>

<VirtualHost *:80>
    DocumentRoot /www/ko1
    ServerName www.ko1.com
    ErrorLog "logs/www.ko1.com-error_log"
    CustomLog "logs/www.ko1.com-access_log" common
        <Directory "/www/ko1">
                Options Indexes FollowSymLinks
                AllowOverride None
                Order allow,deny
                Allow from all
        </Directory>
</VirtualHost>
