#include <stdio.h>

int main(void)
{
        int L;
        int cur;
        int num;
        int count;

        while (scanf("%d", &L) != EOF)
        {
                count = 0;
                while (L--)
                {
                        scanf("%d", &cur);
                        if (count == 0)
                        {
                                num = cur;
                                count++;
                        }
                        else if (cur == num)
                                count++;
                        else
                                count--;
                }
                printf("%d\n", num);
        }
        
        return 0;
}
