function getmms()
{
$pattern='/<a.*(mms[^\"]+).*>(.*?)<\/a>/';
$contents = file_get_contents('file://c:/iptv.htm');

preg_match_all($pattern,$contents, $matches);
//var_dump($matches);
echo "频道数目：".count($matches[0])."<br/>";
for($i=0;$i<count($matches[0]);$i++)
{
echo ($i+1)."<br/>";
echo "播放地址：".$matches[1][$i]."<br/>";
echo "节目名称：".$matches[2][$i]."<br/>";
}
return $matches;
}
