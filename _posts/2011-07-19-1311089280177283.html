#include <stdio.h>
#include <stdlib.h>

#define swap(a, b) {a = a + b; b = a - b; a = a - b;}

int main(int argc, char* argv[])
{
    int a = 3;
    int b = 4;
    printf("%d\t%d\n", a, b);
    swap(a, b);
    printf("%d\t%d\n", a, b);

    return 0;
}
