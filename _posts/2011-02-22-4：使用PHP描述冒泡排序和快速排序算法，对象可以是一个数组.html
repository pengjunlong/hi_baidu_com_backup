 6     function bubble_sort($arr){
 7         $count = count($arr);
 8         if ($count <= 0) return false;
 9         for($i=0; $i<$count; $i++){
10             for($j=$count-1; $j>$i; $j--){
11                  if ($arr[$j] < $arr[$j-1]){
12                      $tmp = $arr[$j];
13                      $arr[$j] = $arr[$j-1];
14                      $arr[$j-1] = $tmp;
15                   }
16               }         }
17          return $arr;
18      }
