Sub 斐波那契()
    Dim wk As Workbook
    Dim wt As Worksheet
    Dim ran As Range
   
    Set wk = Workbooks(1)
    Set wt = wk.Worksheets("sheet2")
    wt.Activate
    Set ran = Range("A1:AJ36")
   
    For i = 1 To 36
        For j = 1 To i
            If j = 1 Then
                ran.Cells(i, j) = 1
            ElseIf j = i Then
                ran.Cells(i, j) = 1
            Else
                ran.Cells(i, j) = ran.Cells(i - 1, j - 1) + ran.Cells(i - 1, j)
            End If
        Next
    Next
   
    Set wk = Nothing
    Set wt = Nothing
    Set ran = Nothing
   
End Sub
