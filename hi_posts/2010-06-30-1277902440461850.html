#include <stdlib.h>
int atexit(void (*function)(void));
#include <stdio.h>

void fn1(void),fn2(void);

int main(void)
{
        atexit(fn1);
        atexit(fn2);
        printf("main\n");
        return 0;
}

void fn1(void)
{
        printf("after:1\n");
}

void fn2(void)
{
        printf("after:2\n");
}
